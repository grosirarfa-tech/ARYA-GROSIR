Cara Menghubungkan Google Spreadsheet:

  1. Buat dokumen **Google Spreadsheet** baru di Google Drive Anda.
  2. Buka menu **Ekstensi** -> **Apps Script**.
  3. Hapus semua kode bawaan, lalu salin dan tempel kode berikut:
function doPost(e) {
  try {
    var payload = JSON.parse(e.postData.contents);
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    
    // 1. Catat Log Backup
    var logSheet = ss.getSheetByName("LOG_BACKUP") || ss.insertSheet("LOG_BACKUP");
    if (logSheet.getLastRow() === 0) {
      logSheet.appendRow(["Waktu Backup", "Jumlah Item Stok", "Jumlah Penjualan", "Jumlah Pengeluaran"]);
    }
    logSheet.appendRow([
      new Date(), 
      payload.inventory.length, 
      payload.sales.length, 
      payload.expenses.length
    ]);
    
    // 2. Simpan Data Stok Terkini
    var stokSheet = ss.getSheetByName("STOK_BARANG") || ss.insertSheet("STOK_BARANG");
    stokSheet.clear();
    stokSheet.appendRow(["Barcode", "Nama Barang", "Sisa Stok", "Harga Beli", "Harga Jual"]);
    payload.inventory.forEach(function(item) {
      stokSheet.appendRow([item.barcode, item.name, item.stock, item.costPrice, item.sellPrice]);
    });
    
    return ContentService.createTextOutput(JSON.stringify({status: "success"}))
      .setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService.createTextOutput(JSON.stringify({status: "error", message: err.toString()}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}
