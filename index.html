Cara Setup Script Database Pusat

  1. Buat Google Spreadsheet baru di Drive Anda.
  2. Klik menu Ekstensi → Apps Script.
  3. Hapus semua kode bawaan, lalu salin-tempel kode Apps Script terpadu berikut:
function doPost(e) {
  try {
    var requestData = JSON.parse(e.postData.contents);
    var action = requestData.action;
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    
    var sheetStok = ss.getSheetByName("STOK_BARANG") || ss.insertSheet("STOK_BARANG");
    var sheetJual = ss.getSheetByName("PENJUALAN") || ss.insertSheet("PENJUALAN");
    var sheetKeluar = ss.getSheetByName("PENGELUARAN") || ss.insertSheet("PENGELUARAN");
    var sheetLog = ss.getSheetByName("LOG_BACKUP") || ss.insertSheet("LOG_BACKUP");
    
    if (action === "push") {
      // 1. Update Stok
      sheetStok.clear();
      sheetStok.appendRow(["Barcode", "Nama Barang", "Sisa Stok", "Harga Beli", "Harga Jual"]);
      if (requestData.stok) {
        requestData.stok.forEach(function(r) {
          sheetStok.appendRow([r.barcode.toString(), r.nama, Number(r.stok), Number(r.hargaBeli), Number(r.hargaJual)]);
        });
      }
      // 2. Update Penjualan
      sheetJual.clear();
      sheetJual.appendRow(["Tanggal", "ID Transaksi", "Detail Barang", "Total Belanja", "Bayar", "Kembalian"]);
      if (requestData.penjualan) {
        requestData.penjualan.forEach(function(r) {
          sheetJual.appendRow([r.tanggal, r.id, r.detail, Number(r.total), Number(r.bayar), Number(r.kembalian)]);
        });
      }
      // 3. Update Pengeluaran
      sheetKeluar.clear();
      sheetKeluar.appendRow(["Tanggal", "Keterangan", "Nominal"]);
      if (requestData.pengeluaran) {
        requestData.pengeluaran.forEach(function(r) {
          sheetKeluar.appendRow([r.tanggal, r.keterangan, Number(r.nominal)]);
        });
      }
      sheetLog.appendRow([new Date(), "Push Backup", "Sukses", (requestData.stok ? requestData.stok.length : 0) + " Item"]);
      return ContentService.createTextOutput(JSON.stringify({ "status": "success", "message": "Backup Berhasil!" })).setMimeType(ContentService.MimeType.JSON);
    } 
    
    else if (action === "pull") {
      // Read Stok
      var stokData = [];
      var sRows = sheetStok.getDataRange().getValues();
      for (var i = 1; i < sRows.length; i++) {
        if (sRows[i][0] !== "") {
          stokData.push({ barcode: sRows[i][0].toString(), nama: sRows[i][1], stok: Number(sRows[i][2])||0, hargaBeli: Number(sRows[i][3])||0, hargaJual: Number(sRows[i][4])||0 });
        }
      }
      // Read Penjualan
      var penjualanData = [];
      var jRows = sheetJual.getDataRange().getValues();
      for (var j = 1; j < jRows.length; j++) {
        if (jRows[j][1] !== "") {
          penjualanData.push({ tanggal: jRows[j][0], id: jRows[j][1], detail: jRows[j][2], total: Number(jRows[j][3])||0, bayar: Number(jRows[j][4])||0, kembalian: Number(jRows[j][5])||0 });
        }
      }
      // Read Pengeluaran
      var pengeluaranData = [];
      var kRows = sheetKeluar.getDataRange().getValues();
      for (var k = 1; k < kRows.length; k++) {
        if (kRows[k][1] !== "") {
          pengeluaranData.push({ tanggal: kRows[k][0], keterangan: kRows[k][1], nominal: Number(kRows[k][2])||0 });
        }
      }
      sheetLog.appendRow([new Date(), "Pull Sync", "Sukses", stokData.length + " Item"]);
      return ContentService.createTextOutput(JSON.stringify({ "status": "success", "stok": stokData, "penjualan": penjualanData, "pengeluaran": pengeluaranData })).setMimeType(ContentService.MimeType.JSON);
    }
  } catch (err) {
    return ContentService.createTextOutput(JSON.stringify({ "status": "error", "message": err.toString() })).setMimeType(ContentService.MimeType.JSON);
  }
}

4. Klik Deploy → New deployment. Atur jenis ke "Web app", akses ke "Anyone", lalu salin URL deployment dan simpan di form URL atas.
