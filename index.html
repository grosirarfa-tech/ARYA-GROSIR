Auto-Backup Aktif: Setiap jam 23:00 sistem akan mencadangkan seluruh data secara otomatis apabila browser ini terbuka.

Panduan Pengaturan Google Sheets

  1. Buat Spreadsheet baru di Google Sheets.
  2. Pastikan untuk membuat sheet/tab dengan nama berikut: STOCK_LIST, PROJECTION, PROFIT, EXPENSES, PURCHASES, dan SALES.
  3. Klik menu Ekstensi > Apps Script.
  4. Hapus semua kode bawaan lalu tempel kode script di bawah ini.
  5. Klik tombol Simpan (ikon disket), lalu terapkan sebagai Aplikasi Web untuk semua orang.

function doPost(e) {
  var lock = LockService.getScriptLock();
  lock.tryLock(10000);
  try {
    var requestData = JSON.parse(e.postData.contents);
    var action = requestData.action;
    var data = requestData.data;
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    
    if (action === "backup") {
      // 1. Backup Catalog (Product Stock)
      var sheetStock = ss.getSheetByName("STOCK_LIST") || ss.insertSheet("STOCK_LIST");
      sheetStock.clear();
      sheetStock.appendRow(["KODE/BARCODE", "NAMA BARANG", "STOCK", "SATUAN", "TARGET MINIMAL BARANG", "HARGA POKOK", "HARGA JUAL"]);
      if (data.products && data.products.length > 0) {
        data.products.forEach(function(p) {
          sheetStock.appendRow([p.barcode, p.name, p.stock, p.unit, p.minStock, p.costPrice, p.sellPrice]);
        });
      }

      // 2. Backup Projection (Perkiraan Belanja)
      var sheetProj = ss.getSheetByName("PROJECTION") || ss.insertSheet("PROJECTION");
      sheetProj.clear();
      sheetProj.appendRow(["NAMA BARANG", "STOCK SAAT INI", "TARGET STOCK", "HARUS DI BELI", "SATUAN", "HARGA POKOK", "SUB TOTAL BELANJA", "TOTAL ANGGARANG BELANJA"]);
      if (data.projections && data.projections.length > 0) {
        data.projections.forEach(function(pr) {
          sheetProj.appendRow([pr.name, pr.stock, pr.minStock, pr.qtyToBuy, pr.unit, pr.costPrice, pr.subtotal, data.projectionTotal]);
        });
      }

      // 3. Backup Profit (Keuntungan)
      var sheetProfit = ss.getSheetByName("PROFIT") || ss.insertSheet("PROFIT");
      sheetProfit.clear();
      sheetProfit.appendRow(["LABA HARI INI", "LABA BULAN INI", "PENGELUARAN PERHARI", "PENGELUARAN PER BULAN", "LABA KOTOR"]);
      if (data.profit) {
        sheetProfit.appendRow([data.profit.labaHariIni, data.profit.labaBulanIni, data.profit.expHariIni, data.profit.expBulanIni, data.profit.labaKotor]);
      }

      // 4. Backup Expenses (Pengeluaran Lain)
      var sheetExp = ss.getSheetByName("EXPENSES") || ss.insertSheet("EXPENSES");
      sheetExp.clear();
      sheetExp.appendRow(["TANGGAL", "NOMINAL PENGELUARAN", "KETERANGAN"]);
      if (data.expenses && data.expenses.length > 0) {
        data.expenses.forEach(function(ex) {
          sheetExp.appendRow([ex.date, ex.amount, ex.desc]);
        });
      }

      // 5. Backup Purchases (Belanja Barang)
      var sheetPurchases = ss.getSheetByName("PURCHASES") || ss.insertSheet("PURCHASES");
      sheetPurchases.clear();
      sheetPurchases.appendRow(["ID TRANSAKSI", "TANGGAL BELANJA", "KODE/BARCODE", "NAMA BARANG", "HARGA PER ITEM", "TOTAL HARAGA"]);
      if (data.purchases && data.purchases.length > 0) {
        data.purchases.forEach(function(pu) {
          pu.items.forEach(function(item) {
            sheetPurchases.appendRow([pu.id, pu.date, item.barcode, item.name, item.costPrice, pu.total]);
          });
        });
      }

      // 6. Backup Sales (Penjualan)
      var sheetSales = ss.getSheetByName("SALES") || ss.insertSheet("SALES");
      sheetSales.clear();
      sheetSales.appendRow(["ID TRANSAKSI", "NAMA PEMBELI", "TANGGAL", "KODE/BARCODE", "NAMA BARANG", "HARGA PER ITEM", "TOTAL HARAGA", "DIBAYAR", "KEMBALIAN"]);
      if (data.sales && data.sales.length > 0) {
        data.sales.forEach(function(s) {
          s.items.forEach(function(item) {
            sheetSales.appendRow([s.id, s.buyer, s.date, item.barcode, item.name, item.sellPrice, s.total, s.paid, s.change]);
          });
        });
      }

      return ContentService.createTextOutput(JSON.stringify({status: "success", message: "Data berhasil dicadangkan!"})).setMimeType(ContentService.MimeType.JSON);
    }
  } catch (err) {
    return ContentService.createTextOutput(JSON.stringify({status: "error", message: err.toString()})).setMimeType(ContentService.MimeType.JSON);
  } finally {
    lock.releaseLock();
  }
}

function doGet(e) {
  try {
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    var sheetStock = ss.getSheetByName("STOCK_LIST");
    var sheetExp = ss.getSheetByName("EXPENSES");
    var sheetPurchases = ss.getSheetByName("PURCHASES");
    var sheetSales = ss.getSheetByName("SALES");
    
    var output = { products: [], expenses: [], purchases: [], sales: [] };
    
    // Read Stock
    if (sheetStock) {
      var rows = sheetStock.getDataRange().getValues();
      for (var i = 1; i < rows.length; i++) {
        output.products.push({
          id: "PROD-" + i,
          barcode: String(rows[i][0]),
          name: String(rows[i][1]),
          stock: Number(rows[i][2]),
          unit: String(rows[i][3]),
          minStock: Number(rows[i][4]),
          costPrice: Number(rows[i][5]),
          sellPrice: Number(rows[i][6])
        });
      }
    }
    
    // Read Expenses
    if (sheetExp) {
      var rowsExp = sheetExp.getDataRange().getValues();
      for (var i = 1; i < rowsExp.length; i++) {
        output.expenses.push({
          id: "EXP-" + i,
          date: String(rowsExp[i][0]),
          amount: Number(rowsExp[i][1]),
          desc: String(rowsExp[i][2])
        });
      }
    }

    // Read Purchases (grouped by ID)
    if (sheetPurchases) {
      var rowsPu = sheetPurchases.getDataRange().getValues();
      var purchaseMap = {};
      for (var i = 1; i < rowsPu.length; i++) {
        var id = String(rowsPu[i][0]);
        var date = String(rowsPu[i][1]);
        var barcode = String(rowsPu[i][2]);
        var name = String(rowsPu[i][3]);
        var cost = Number(rowsPu[i][4]);
        var total = Number(rowsPu[i][5]);
        
        if (!purchaseMap[id]) {
          purchaseMap[id] = { id: id, date: date, total: total, items: [] };
        }
        purchaseMap[id].items.push({ barcode: barcode, name: name, qty: 1, unit: "pcs", costPrice: cost });
      }
      for (var key in purchaseMap) {
        output.purchases.push(purchaseMap[key]);
      }
    }
    
    // Read Sales (grouped by ID)
    if (sheetSales) {
      var rowsSales = sheetSales.getDataRange().getValues();
      var salesMap = {};
      for (var i = 1; i < rowsSales.length; i++) {
        var id = String(rowsSales[i][0]);
        var buyer = String(rowsSales[i][1]);
        var date = String(rowsSales[i][2]);
        var barcode = String(rowsSales[i][3]);
        var name = String(rowsSales[i][4]);
        var price = Number(rowsSales[i][5]);
        var total = Number(rowsSales[i][6]);
        var paid = Number(rowsSales[i][7]);
        var change = Number(rowsSales[i][8]);
        
        if (!salesMap[id]) {
          salesMap[id] = { id: id, buyer: buyer, date: date, total: total, paid: paid, change: change, items: [] };
        }
        salesMap[id].items.push({ barcode: barcode, name: name, qty: 1, unit: "pcs", sellPrice: price });
      }
      for (var key in salesMap) {
        output.sales.push(salesMap[key]);
      }
    }
    
    return ContentService.createTextOutput(JSON.stringify({status: "success", data: output})).setMimeType(ContentService.MimeType.JSON);
  } catch (err) {
    return ContentService.createTextOutput(JSON.stringify({status: "error", message: err.toString()})).setMimeType(ContentService.MimeType.JSON);
  }
}
