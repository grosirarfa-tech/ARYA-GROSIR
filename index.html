Panduan Pengaturan Google Sheets

  1. Buat Spreadsheet baru di Google Sheets.
  2. Klik menu Ekstensi > Apps Script.
  3. Hapus semua kode bawaan lalu tempel kode script di bawah ini.
  4. Klik tombol Simpan (ikon disket).
  5. Klik Terapkan > Penerapan Baru. Pilih jenis Aplikasi Web.
  6. Isi Deskripsi, setel Akses sebagai: Anda dan Siapa saja yang memiliki akses: Siapa Saja.
  7. Klik Terapkan, beri otorisasi, lalu salin URL Web App yang dihasilkan ke form di sebelah kiri.

/* KODE GOOGLE APPS SCRIPT DATABASE ARYA GROSIR */
function doPost(e) {
  var lock = LockService.getScriptLock();
  lock.tryLock(10000);
  try {
    var requestData = JSON.parse(e.postData.contents);
    var action = requestData.action;
    var data = requestData.data;
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    
    if (action === "backup") {
      // Backup Catalog (Product Stock)
      var sheetStock = ss.getSheetByName("STOCK_LIST") || ss.insertSheet("STOCK_LIST");
      sheetStock.clear();
      sheetStock.appendRow(["ID", "Barcode", "Nama Barang", "Stock", "Unit", "Min Stock", "Cost Price", "Sell Price"]);
      if (data.products && data.products.length > 0) {
        data.products.forEach(function(p) {
          sheetStock.appendRow([p.id, p.barcode, p.name, p.stock, p.unit, p.minStock, p.costPrice, p.sellPrice]);
        });
      }

      // Backup Expenses
      var sheetExp = ss.getSheetByName("EXPENSES") || ss.insertSheet("EXPENSES");
      sheetExp.clear();
      sheetExp.appendRow(["ID", "Tanggal", "Keterangan", "Nominal"]);
      if (data.expenses && data.expenses.length > 0) {
        data.expenses.forEach(function(ex) {
          sheetExp.appendRow([ex.id, ex.date, ex.desc, ex.amount]);
        });
      }

      // Backup Sales
      var sheetSales = ss.getSheetByName("SALES") || ss.insertSheet("SALES");
      sheetSales.clear();
      sheetSales.appendRow(["ID Transaksi", "Tanggal", "Pembeli", "Item Terjual", "Total", "Dibayar", "Kembalian"]);
      if (data.sales && data.sales.length > 0) {
        data.sales.forEach(function(s) {
          sheetSales.appendRow([s.id, s.date, s.buyer, JSON.stringify(s.items), s.total, s.paid, s.change]);
        });
      }
      
      // Backup Purchases (Belanja)
      var sheetPurchases = ss.getSheetByName("PURCHASES") || ss.insertSheet("PURCHASES");
      sheetPurchases.clear();
      sheetPurchases.appendRow(["ID Belanja", "Tanggal", "Item Dibeli", "Total"]);
      if (data.purchases && data.purchases.length > 0) {
        data.purchases.forEach(function(pu) {
          sheetPurchases.appendRow([pu.id, pu.date, JSON.stringify(pu.items), pu.total]);
        });
      }

      return ContentService.createTextOutput(JSON.stringify({status: "success", message: "Data berhasil dicadangkan!"})).setMimeType(ContentService.MimeType.JSON);
    }
  } catch (err) {
    return ContentService.createTextOutput(JSON.stringify({status: "error", message: err.toString()})).setMimeType(ContentService.MimeType.JSON);
  } finally {
    lock.releaseLock();
  }
}

function doGet(e) {
  try {
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    var sheetStock = ss.getSheetByName("STOCK_LIST");
    var sheetExp = ss.getSheetByName("EXPENSES");
    var sheetSales = ss.getSheetByName("SALES");
    var sheetPurchases = ss.getSheetByName("PURCHASES");
    
    var output = { products: [], expenses: [], sales: [], purchases: [] };
    
    // Read Stock
    if (sheetStock) {
      var rows = sheetStock.getDataRange().getValues();
      for (var i = 1; i < rows.length; i++) {
        output.products.push({
          id: rows[i][0], barcode: rows[i][1], name: rows[i][2],
          stock: Number(rows[i][3]), unit: rows[i][4], minStock: Number(rows[i][5]),
          costPrice: Number(rows[i][6]), sellPrice: Number(rows[i][7])
        });
      }
    }
    
    // Read Expenses
    if (sheetExp) {
      var rowsExp = sheetExp.getDataRange().getValues();
      for (var i = 1; i < rowsExp.length; i++) {
        output.expenses.push({ id: rowsExp[i][0], date: rowsExp[i][1], desc: rowsExp[i][2], amount: Number(rowsExp[i][3]) });
      }
    }

    // Read Sales
    if (sheetSales) {
      var rowsSales = sheetSales.getDataRange().getValues();
      for (var i = 1; i < rowsSales.length; i++) {
        output.sales.push({
          id: rowsSales[i][0], date: rowsSales[i][1], buyer: rowsSales[i][2],
          items: JSON.parse(rowsSales[i][3] || "[]"), total: Number(rowsSales[i][4]),
          paid: Number(rowsSales[i][5]), change: Number(rowsSales[i][6])
        });
      }
    }
    
    // Read Purchases
    if (sheetPurchases) {
      var rowsPu = sheetPurchases.getDataRange().getValues();
      for (var i = 1; i < rowsPu.length; i++) {
        output.purchases.push({
          id: rowsPu[i][0], date: rowsPu[i][1],
          items: JSON.parse(rowsPu[i][2] || "[]"), total: Number(rowsPu[i][3])
        });
      }
    }
    
    return ContentService.createTextOutput(JSON.stringify({status: "success", data: output})).setMimeType(ContentService.MimeType.JSON);
  } catch (err) {
    return ContentService.createTextOutput(JSON.stringify({status: "error", message: err.toString()})).setMimeType(ContentService.MimeType.JSON);
  }
}
