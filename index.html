Pengaturan Web App Google Apps Script

2-Way Sync Backend

Masukkan URL Web App hasil deploy Apps Script Google Sheets Anda di sini.

Panduan Setup Backend & Kode Google Apps Script

  1. Buka Spreadsheet Google Sheets Baru di Google Drive Anda.
  2. Klik menu Extensions > Apps Script.
  3. Hapus semua isi kode default, lalu salin dan tempelkan (copy-paste) kode Apps Script di bawah ini:
function doGet() {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var result = {};
  var sheets = ['KATALOG', 'BELANJA', 'PENJUALAN', 'PENGELUARAN'];
  sheets.forEach(function(s) {
    var sheet = ss.getSheetByName(s);
    if (!sheet) {
      result[s] = [];
    } else {
      var data = sheet.getDataRange().getValues();
      result[s] = data;
    }
  });
  return ContentService.createTextOutput(JSON.stringify(result)).setMimeType(ContentService.MimeType.JSON);
}

function doPost(e) {
  try {
    var payload = JSON.parse(e.postData.contents);
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    
    // Save or update sheets
    for (var key in payload) {
      var sheet = ss.getSheetByName(key) || ss.insertSheet(key);
      sheet.clearContents();
      if (payload[key] && payload[key].length > 0) {
        sheet.getRange(1, 1, payload[key].length, payload[key][0].length).setValues(payload[key]);
      }
    }
    
    // Auto Backup Timestamp Log
    var logSheet = ss.getSheetByName('LOG_BACKUP') || ss.insertSheet('LOG_BACKUP');
    logSheet.appendRow([new Date(), 'Auto-Backup Sync Success']);
    
    return ContentService.createTextOutput(JSON.stringify({status: 'success'})).setMimeType(ContentService.MimeType.JSON);
  } catch (err) {
    return ContentService.createTextOutput(JSON.stringify({status: 'error', message: err.toString()})).setMimeType(ContentService.MimeType.JSON);
  }
}

4. Klik Deploy > New deployment.

5. Pilih type: Web app. Set Execute as: Me dan Who has access: Anyone.

6. Salin URL Web App dan tempel pada kolom input di atas!
