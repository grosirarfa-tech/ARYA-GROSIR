Cara Setup Script Database Pusat

  1. Buat Google Spreadsheet baru di Drive Anda.
  2. Klik menu Ekstensi → Apps Script.
  3. Hapus kode bawaan, lalu salin-tempel kode Apps Script terpadu di bawah ini:
function doPost(e) {
  var token = "ARYA_GROSIR_SECURE_SYNC";
  var contents = JSON.parse(e.postData.contents);
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  
  // 1. STOK BARANG
  var sheetStok = ss.getSheetByName("STOK_BARANG") || ss.insertSheet("STOK_BARANG");
  sheetStok.clear();
  sheetStok.appendRow(["Barcode", "Nama Barang", "Sisa Stok", "Harga Beli", "Harga Jual"]);
  contents.stok.forEach(function(r) {
    sheetStok.appendRow([r.barcode, r.nama, r.stok, r.hargaBeli, r.hargaJual]);
  });
  
  // 2. TRANSAKSI PENJUALAN
  // ... (lanjutan script sync Anda)
}
